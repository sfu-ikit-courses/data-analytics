from scipy import stats


def calculate_statistics(sample, name):
    print(f"   {name}:")
    print(f": {np.min(sample)}")
    print(f": {np.max(sample)}")
    print(f" : {np.mean(sample)}")
    print(f": {np.var(sample)}")
    print(f" : {np.std(sample)}")
    print(f" 0.25: {np.quantile(sample, 0.25)}")
    print(f" 0.75: {np.quantile(sample, 0.75)}")
    print(f": {np.ptp(sample)}")
    print(f": {np.median(sample)}")
    print(f" : {stats.skew(sample)}")
    print(f" : {stats.kurtosis(sample)}")
